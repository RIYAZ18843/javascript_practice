// Define a class named 'Car' class Car { // Constructor function to initialize object properties constructor(make, model, year) { this.make = make; // Property: car make (e.g., Toyota) this.model = model; // Property: car model (e.g., Camry) this.year = year; // Property: car manufacture year (e.g., 2020) } // Method to display car information displayInfo() { console.log('Car: ' + this.year + ' ' + this.make + ' ' + this.model); } } // Creating instances (objects) of the Car class let myCar = new Car('Toyota', 'Camry', 2020); let anotherCar = new Car('Honda', 'Civic', 2021); // Using the objects and their methods myCar.displayInfo(); // Output: Car: 2020 Toyota Camry anotherCar.displayInfo(); // Output: Car: 2021 Honda Civic
